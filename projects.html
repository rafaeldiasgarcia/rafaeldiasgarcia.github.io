async function createBook(req, res) {
    const { title, author, year } = req.body;
    const newBook = {
        id: uuidv4(),
        title,
        author,
        year: parseInt(year)
    };
    books.push(newBook);
    await writeData(books);
    res.status(201).json(newBook);
}

async function deleteBook(req, res) {
    books = books.filter(b => b.id !== req.params.id);
    await writeData(books);
    res.status(204).send();
}
