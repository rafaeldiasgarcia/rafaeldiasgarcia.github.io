EventService.php 
public function scheduleEvent(CreateEventDTO $dto): Event
{
    return DB::transaction(function () use ($dto) {
        // Check availability for the time slot
        if (!$this->repo->isTimeSlotAvailable($dto->date, $dto->time)) {
             throw new ConflictException("Horário indisponível.");
        }

        // Create event with RBAC protection
        $event = $this->repo->create($dto->toArray());

        // Notify admins asynchronously
        $this->notificationService->notifyNewSchedule($event);

        return $event;
    });
}
